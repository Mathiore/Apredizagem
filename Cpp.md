#include <iostream>

using namespace std;


int main() {
    setlocale(LC_ALL, "Portuguese");
    int distancia, consumo, litros;
    float custo, gasto;
    cout<<"Distância da viagem em km?";

    cout<<"Quanto é o custo da gasolina?";

}
